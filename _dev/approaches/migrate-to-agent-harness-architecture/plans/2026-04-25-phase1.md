**接口定义**：

```typescript
import { AgentSession } from "@mariozechner/pi-coding-agent";

export interface AltTheoryConfig {
  /** KB 根目录 */
  kbDir: string;
  /** Profile 文件路径（可选） */
  profilePath?: string;
  /** 是否只读模式 */
  readOnly: boolean;
}

export async function createAltTheorySession(
  config: AltTheoryConfig
): Promise<{ session: AgentSession }>;
```

**实现要点**：
1. 读取 `profilePath`（如有），拼接到 appendSystemPromptOverride
2. 追加 KB 路径声明
3. 使用 `DefaultResourceLoader({ appendSystemPromptOverride })`
4. 根据 `readOnly` 选择工具：`createReadOnlyTools(kbDir)` 或 `createCodingTools(kbDir)`
5. 使用 `SessionManager.create()` 或 `SessionManager.inMemory()`

**⚠️ 重要注意事项（Opus review）**：
1. **API 签名确认**：`appendSystemPromptOverride` 的类型签名不确定（函数 vs 字符串数组）。写代码前必须先查 `DefaultResourceLoader` 的 TypeScript 定义（`node_modules/@mariozechner/pi-coding-agent/dist/index.d.ts`）。
2. **cwd 设置**：`cwd` 应是 assets/ 的父目录（包含 kb/、profiles/、sessions/），**不是** `kbDir`。`kbDir` 作为搜索路径单独传给 `createReadOnlyTools(kbDir)`。当前 plan 中的 `const cwd = config.kbDir` 是 bug，dev session 需要修正。

**Steps:**

- [ ] **Step 1: 写 Core 层代码**

参考代码框架：
```typescript
import {
  createAgentSession,
  DefaultResourceLoader,
  getAgentDir,
  SessionManager,
  createReadOnlyTools,
  createCodingTools,
} from "@mariozechner/pi-coding-agent";
import { readFileSync, existsSync } from "fs";

export async function createAltTheorySession(config: AltTheoryConfig) {
  // ⚠️ 修正：cwd 应是包含 kb/ profiles/ sessions/ 的父目录，不是 kbDir 本身
  const cwd = process.cwd(); // 或从 config 传入的 rootDir
  const kbDir = config.kbDir;
  const agentDir = getAgentDir();
  
  // 读取 profile
  const profileContent = config.profilePath && existsSync(config.profilePath)
    ? readFileSync(config.profilePath, "utf-8")
    : "";
  
  // 组装 appendSystemPromptOverride
  const appendContent = [
    profileContent && `## User Profile\n${profileContent}`,
    `## Knowledge Base\nYour knowledge base is at: ${config.kbDir}`,
  ].filter(Boolean);
  
  const loader = new DefaultResourceLoader({
    cwd,
    agentDir,
    appendSystemPromptOverride: (base) => [...base, ...appendContent],
  });
  await loader.reload();
  
  // ⚠️ 修正：工具绑定 kbDir，不是 cwd
  const tools = config.readOnly
    ? createReadOnlyTools(kbDir)
    : createCodingTools(kbDir);
  
  const { session } = await createAgentSession({
    cwd,
    resourceLoader: loader,
    sessionManager: SessionManager.inMemory(),
    tools,
  });
  
  return { session };
}
```
